  <script>

    // 1) EDIT YOUR LINKS HERE

    const APPS = [

      {

        name: "OT Session Registry (HPG)",

        url: "https://YOUR_WEBAPP_URL_HERE",

        category: "OT / Theatre",

        desc: "Create sessions, add cases, finalise, print PDF list.",

        pinned: true,

        tag: "Most used"

      },

      {

        name: "Appointment Booking System",

        url: "https://YOUR_WEBAPP_URL_HERE",

        category: "Clinic Ops",

        desc: "Room booking with staggered slots and lunch breaks.",

        pinned: true,

        tag: "Daily"

      },

      {

        name: "Inventory / Stock Request",

        url: "https://YOUR_WEBAPP_URL_HERE",

        category: "Admin",

        desc: "Track items, requests, and restock status.",

        pinned: false

      },

      {

        name: "Training / Quiz Portal",

        url: "https://YOUR_WEBAPP_URL_HERE",

        category: "Education",

        desc: "OMFS MCQ, timer, scoring, and question bank.",

        pinned: false

      }

      // Add more objects here...

    ];


    // 2) RENDER LOGIC (no need to edit)

    const $apps = document.getElementById("apps");

    const $pinned = document.getElementById("pinned");

    const $q = document.getElementById("q");

    const $cat = document.getElementById("cat");

    const $pinnedBlock = document.getElementById("pinnedBlock");


    function uniq(arr){ return [...new Set(arr)].sort(); }


    function card(app){

      const el = document.createElement("div");

      el.className = "card";

      el.dataset.name = app.name.toLowerCase();

      el.dataset.desc = (app.desc || "").toLowerCase();

      el.dataset.category = app.category;


      el.innerHTML = `

        <div class="topline">

          <p class="name">${app.name}</p>

          ${app.tag ? `<span class="badge">${app.tag}</span>` : ``}

        </div>

        <p class="desc">${app.desc || ""}</p>

        <a class="btn" href="${app.url}" target="_blank" rel="noopener">Open</a>

        <div class="meta">${app.category || "General"}</div>

      `;

      return el;

    }


    function populateCategories(){

      const cats = uniq(APPS.map(a => a.category || "General"));

      cats.forEach(c => {

        const opt = document.createElement("option");

        opt.value = c; opt.textContent = c;

        $cat.appendChild(opt);

      });

    }


    function render(){

      $apps.innerHTML = "";

      $pinned.innerHTML = "";


      const query = ($q.value || "").trim().toLowerCase();

      const cat = $cat.value;


      let pinnedCount = 0;


      APPS.forEach(app => {

        const matchesText =

          !query ||

          app.name.toLowerCase().includes(query) ||

          (app.desc || "").toLowerCase().includes(query) ||

          (app.category || "").toLowerCase().includes(query);


        const matchesCat = (cat === "all") || (app.category === cat);


        if (!matchesText || !matchesCat) return;


        const el = card(app);


        if (app.pinned){

          $pinned.appendChild(el);

          pinnedCount++;

        } else {

          $apps.appendChild(el);

        }

      });


      $pinnedBlock.classList.toggle("hidden", pinnedCount === 0);

    }


    populateCategories();

    render();


    $q.addEventListener("input", render);

    $cat.addEventListener("change", render);


    document.getElementById("updated").textContent =

      new Date().toLocaleString(undefined, { year:"numeric", month:"short", day:"2-digit" });
